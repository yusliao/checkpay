1.1 设计决策

  -------------------------------------------------------------------------------------------------------------
  ① 时区：所有时间字段统一存储 UTC（timestamptz）。前端按用户所在地转换显示（美国 UTC-5\~8 / 大陆 UTC+8）。

  ② 主键：全表使用 UUID v4，避免分布式环境下 ID 冲突，同时不暴露业务序号。

  ③ 软删除：核心业务表不物理删除，使用 deleted_at 字段标记。金融数据需完整保留审计链。

  ④ OCR 临时数据：OCR 原始结果单独存表（ocr_results），与正式业务记录解耦。确认入库后关联，原始数据永久保留。

  ⑤ 乐观并发：check_records 和 debit_records 表加 row_version 字段，防止多人同时修改同一记录冲突。

  ⑥ JSONB 扩展字段：OCR 原始 JSON 存 JSONB，便于未来扩展字段而无需 migrate。
  -------------------------------------------------------------------------------------------------------------

1.2 表清单总览

  -----------------------------------------------------------------------
  表名中文名说明
  ---------------- ---------------- -------------------------------------
  users            用户表           账号、角色、所属地区

  customers        客户主数据表     客户编号、名称等

  check_records    支票记录表       流程一：支票采集结果（核心表）

  debit_records    扣款记录表       流程二：银行扣款信息

  ocr_results      OCR结果表        两个流程的OCR原始输出，与业务表解耦

  audit_logs       审计日志表       所有状态变更、关键操作的完整日志

  import_batches   导入批次表       每次文件上传的批次记录

2.1 收款记录状态（record_status）

  ------------------------------------------------------------------------------
        枚举值显示名称含义与流转条件
  ----- ---------------- -------------- ----------------------------------------
  🟡    pending_debit    待扣款     支票采集完成，等待美国财务导入扣款记录

  🔴    anomaly          异常       扣款导入时支票号匹配失败，需手动关联

  🔵    pending_review   待核查     扣款匹配成功，等待大陆财务核查

  🟠    disputed         存疑       核查发现问题，必须填写存疑原因

  🟢    confirmed        已确认     核查通过，流程结束（可回退至存疑）

3.3 check_records --- 支票记录表（核心）

  -----------------------------------------------------------------------
  📌 这是系统的核心表。每条记录对应一张支票，贯穿完整业务生命周期。

  status 字段驱动整个状态机，所有状态变更同步写入 audit_logs。
  -----------------------------------------------------------------------

+------------------------------------------------------------------------------------------+
| CREATE TABLE check_records (                                                             |
|                                                                                          |
| -- 主键与基础信息                                                                       |
|                                                                                          |
| id uuid PRIMARY KEY DEFAULT gen_random_uuid(),                                           |
|                                                                                          |
| check_number varchar(50) NOT NULL, -- ⚠️ 匹配关键字段                                   |
|                                                                                          |
| amount decimal(12,2) NOT NULL,                                                           |
|                                                                                          |
| check_date date NOT NULL, -- 支票上的日期                                               |
|                                                                                          |
| customer_id uuid NOT NULL REFERENCES customers(id),                                      |
|                                                                                          |
| image_url text NOT NULL, -- Azure Blob Storage SAS URL                                  |
|                                                                                          |
| image_blob_path text NOT NULL, -- Blob 存储路径（重新生成SAS用）                        |
|                                                                                          |
| notes text,                                                                              |
|                                                                                          |
| -- OCR 关联                                                                             |
|                                                                                          |
| ocr_result_id uuid REFERENCES ocr_results(id),                                           |
|                                                                                          |
| -- 状态机                                                                               |
|                                                                                          |
| status record_status NOT NULL DEFAULT \'pending_debit\',                                 |
|                                                                                          |
| -- 存疑原因（status = disputed 时必填）                                                 |
|                                                                                          |
| dispute_reason text,                                                                     |
|                                                                                          |
| -- 关联扣款记录（匹配成功后填入）                                                       |
|                                                                                          |
| debit_record_id uuid REFERENCES debit_records(id),                                       |
|                                                                                          |
| -- 审计字段                                                                             |
|                                                                                          |
| created_by uuid NOT NULL REFERENCES users(id),                                           |
|                                                                                          |
| created_at timestamptz NOT NULL DEFAULT now(),                                           |
|                                                                                          |
| updated_at timestamptz NOT NULL DEFAULT now(),                                           |
|                                                                                          |

| );                                                                                       |
|                                                                                          |

|                                                                                          |

|                                                                                          |

|                                                                                          |
| CREATE UNIQUE INDEX idx_check_number_active                                              |
|                                                                                          |
| ON check_records(check_number) WHERE deleted_at IS NULL;                                 |

  ----------------------------------------------------------------------------------------------
  字段类型说明
  ------------------- ------------- ------------------------------------------------------------
  check_number        varchar(50)   唯一约束（软删除范围内），支票号是流程二自动匹配的唯一键

  image_blob_path     text          存储原始路径，image_url 是有时效的 SAS URL，需要时重新生成

  dispute_reason      text          数据库层 CHECK 约束：status = disputed 时必须有值

  row_version         integer       乐观锁，每次 UPDATE 递增；前端提交时带上，不匹配则拒绝

  debit_record_id     uuid FK       1:1 关系，匹配成功后由系统填入，手动关联时也通过此字段建立

  ocr_result_id       uuid FK       关联 OCR 临时表，确认入库后建立，原始识别数据永久保留

3.4 debit_records --- 扣款记录表

4.1 audit_logs --- 审计日志

  ----------------------------------------------------------------------------
  📌 所有状态变更、关键操作均写入此表。只增不改（append-only），不做软删除。

  写入时机：应用层事务提交后同步写入（与业务操作同一事务）。
  ----------------------------------------------------------------------------

+---------------------------------------------------------------------------------------+
| CREATE TABLE audit_logs (                                                             |
|                                                                                       |
| id uuid PRIMARY KEY DEFAULT gen_random_uuid(),                                        |
|                                                                                       |
| -- 操作者                                                                            |
|                                                                                       |
| operator_id uuid NOT NULL REFERENCES users(id),                                       |
|                                                                                       |
| operator_name varchar(100) NOT NULL, -- 冗余存储，防止用户改名后日志失真             |
|                                                                                       |
| -- 操作类型与对象                                                                    |
|                                                                                       |
| action audit_action NOT NULL,                                                         |
|                                                                                       |
| entity_type varchar(50) NOT NULL, -- \'check_record\' | \'debit_record\' | \...    |
|                                                                                       |
| entity_id uuid NOT NULL,                                                              |
|                                                                                       |
| -- 状态变更（仅 status_changed 时有值）                                              |
|                                                                                       |
| status_before record_status,                                                          |
|                                                                                       |
| status_after record_status,                                                           |
|                                                                                       |
| -- 变更内容快照                                                                      |
|                                                                                       |
| changes jsonb, -- { field: \[old_value, new_value\], \... }                          |
|                                                                                       |
| note text, -- 操作备注（存疑原因等）                                                 |
|                                                                                       |
| -- 时间（服务端 UTC）                                                                |
|                                                                                       |
| created_at timestamptz NOT NULL DEFAULT now()                                         |
|                                                                                       |
| );                                                                                    |
|                                                                                       |
| -- 按实体查询（最常用：查一条记录的操作历史）                                        |
|                                                                                       |
| CREATE INDEX idx_audit_entity ON audit_logs(entity_type, entity_id, created_at DESC); |
|                                                                                       |
| -- 按操作者查询                                                                      |
|                                                                                       |
| CREATE INDEX idx_audit_operator ON audit_logs(operator_id, created_at DESC);          |
|                                                                                       |
| -- 按时间范围查询（报表/合规）                                                       |
|                                                                                       |
| CREATE INDEX idx_audit_created ON audit_logs(created_at DESC);                        |
+---------------------------------------------------------------------------------------+

  ----------------------------------------------------------------------------------------------------------------------
  字段说明
  ------------------ ---------------------------------------------------------------------------------------------------
  operator_name      冗余存储显示名，避免用户 display_name 变更后历史日志对不上

  changes            JSONB 格式 { 字段名: \[旧值, 新值\] }，例如 { \"status\": \[\"pending_review\", \"confirmed\"\] }

  note               存疑时填写的原因、手动关联备注等自由文本

  entity_type        字符串区分实体类型，不用 FK，保证日志表对业务表无强依赖
  ----------------------------------------------------------------------------------------------------------------------

+-----------------------------------------------------------------------+
| 5. 状态流转矩阵                                                   |
|                                                                       |
| State Transition Matrix                                               |
+-----------------------------------------------------------------------+

所有状态变更均可逆（已确认 → 存疑），所有变更写入 audit_logs。

  --------------------------------------------------------------------------------------------------------
  当前状态触发动作下一状态执行人备注
  -------------- -------------------------- -------------- --------------- -------------------------------
  🟡 待扣款      扣款导入，支票号匹配成功   🔵 待核查      系统自动        主路径

  🟡 待扣款      扣款导入，支票号匹配失败   🔴 异常        系统自动        美国财务需处理

  🔴 异常        手动关联扣款记录           🔵 待核查      美国财务        关联后 matched_by 记录操作人

  🔵 待核查      核查无误，点击确认         🟢 已确认      大陆/美国财务   confirmed_by/at 同步写入

  🔵 待核查      发现问题，填写原因         🟠 存疑        大陆/美国财务   dispute_reason 必填（DB约束）

  🟠 存疑        线下沟通后问题解决         🟢 已确认      大陆/美国财务   ---

  🟠 存疑        重新核查，恢复待核查       🔵 待核查      大陆/美国财务   ---

  🟢 已确认      发现确认有误，回退         🟠 存疑        大陆/美国财务   需填写存疑原因
  --------------------------------------------------------------------------------------------------------

  ------------------------------------------------------------------------------------
  ⚠️ 乐观锁规则：所有状态变更 API 必须携带 row_version。

  服务端 UPDATE WHERE id = ? AND row_version = ?，影响行数为 0 时返回 409 Conflict。

  前端收到 409 后提示用户刷新页面重新操作。
  ------------------------------------------------------------------------------------

6.2 收款记录多条件筛选

+------------------------------------------------------------------------------------+
| -- 支持：状态/客户/日期范围/支票号 筛选，分页                                     |
|                                                                                    |
| SELECT                                                                             |
|                                                                                    |
| cr.id, cr.check_number, cr.amount, cr.check_date,                                  |
|                                                                                    |
| cr.status, cr.dispute_reason,                                                      |
|                                                                                    |
| c.customer_code, c.customer_name,                                                  |
|                                                                                    |
| dr.debit_amount, dr.bank_reference, dr.debit_date,                                 |
|                                                                                    |
| u.display_name AS created_by_name                                                  |
|                                                                                    |
| FROM check_records cr                                                              |
|                                                                                    |
| JOIN customers c ON cr.customer_id = c.id                                          |
|                                                                                    |
| JOIN users u ON cr.created_by = u.id                                               |
|                                                                                    |
| LEFT JOIN debit_records dr ON cr.debit_record_id = dr.id                           |
|                                                                                    |
| WHERE cr.deleted_at IS NULL                                                        |
|                                                                                    |
| AND (:status IS NULL OR cr.status = :status::record_status)                        |
|                                                                                    |
| AND (:customer IS NULL OR c.customer_code = :customer)                             |
|                                                                                    |
| AND (:date_from IS NULL OR cr.check_date \>= :date_from)                           |
|                                                                                    |
| AND (:date_to IS NULL OR cr.check_date \<= :date_to)                               |
|                                                                                    |
| AND (:check_num IS NULL OR cr.check_number ILIKE \'%\' || :check_num || \'%\') |
|                                                                                    |
| ORDER BY cr.check_date DESC, cr.created_at DESC                                    |
|                                                                                    |
| LIMIT :page_size OFFSET :offset;                                                   |
